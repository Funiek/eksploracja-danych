def cov(x, y, x_avg, y_avg):
    elements_count = len(x) # ilo elementw w tablicy
    elements_sum = 0
    
    for x,y in zip(x,y):
        elements_sum = elements_sum + ((x - x_avg)*(y - y_avg)) # sumowanie (x - rednia_arytmetyczna_x) * (y - rednia_arytmetyczna_y)
    
    return elements_sum / elements_count # suma dzielona przez liczb elementw

def cov_for_dataset(cov_dataset):
    avgs = []
    matrix_shape = cov_dataset.shape[1] # wymiary_macierzy czyli liczba_kolumn_bioru_danych x liczba_kolumn_bioru_danych
    for i in range(matrix_shape):
        avgs.append(cov_dataset[:,i].mean()) # obliczanie rednich dla kadego parametru
    
    matrix = []
    for i in range(matrix_shape):
        matrix_row = []
        for j in range(matrix_shape):
            #matrix_shape*i + j
            res = cov(cov_dataset[:,i],cov_dataset[:,j],avgs[i], avgs[j]) # wyznaczanie kowariancji dla kadej pary parametrw
            matrix_row.append(res) # budowanie wiersza macierzy
        matrix.append(matrix_row) # dodawanie wiersza macierzy do tablicy bazowej
    
    return np.matrix(matrix) # przeksztacanie typu tablicy na macierz

def eigenvector_len(v1):
    dims = v1.shape[0] # ilo elementw wektora
    length = 0
    
    for i in range(dims):
        val = v1[i].ravel().tolist()[0][0]
        length = length + pow(val,2) # sumowanie elementw podniesionych do kwadratu
        
    
    length = sqrt(length) # pierwiastkowanie wyniku sumy elementw wektora podniesionych do kwadratw
    
    return length # zwracanie dugoci

Funkcja normalizujca, zwracajca warto wasn i wektor wasny

def eigenvector_normalization(v1):
    eigenvalue = eigenvector_len(v1) # wylicznaie wartoci wasnej
    eigenvector = v1 / eigenvalue # wyliczanie wektora wasnego
    
    return eigenvalue, eigenvector # zwrcenie wartoci wasnej i wektora wasnego

def rearrange_matrix(eigenvectors):
    res = []

    for i in range(len(eigenvectors)):
        row = eigenvectors[i].ravel().tolist()[0]
        res.append(row)

    return np.matrix(res).T

def eigenvalues_eigenvectors(matrix, num_iterations):
    n = matrix.shape[0] # jeden z wymiarw macierzy, poniewa jest kwadratowa to wystarczy tylko jeden
    eigenvalues = [] # lista na wartoci wasne
    eigenvectors = [] # lista na wektory wasne
    
    for i in range(n):
        # losowe dane wektora na pocztku
        eigenvector = np.random.rand(n,1)
    
        # algorytm iteracyjny wykonywany num_iterations razy
        for _ in range(num_iterations):
            eigenvector = np.dot(matrix, eigenvector) # mnoenie macierzy kowariancji przez wektor 
            eigenvalue, eigenvector = eigenvector_normalization(eigenvector) # normalizacja wektora
            
        # usuwanie z macierzy znalezionej wartoci wasnej i wektora wasnego
        # marix - warto wasna * iloczyn zewntrzny wektora wasnego
        matrix = matrix - eigenvalue * np.multiply(eigenvector,eigenvector.T).T 
        
        eigenvalues.append(eigenvalue) # dodanie znalezionej wartoci wasnej do listy
        eigenvectors.append(eigenvector) # dodanie znalezionego wektora wasnego do listy
        
        
        
    return eigenvalues, rearrange_matrix(eigenvectors) # zwrcenie wartoci wasnych i wektorw wasnych

def draw_plot(datasets):
    colors = ['red','blue','green']
    
    print(len(datasets))
    
    # dla kadego zbioru danych z danej kategorii
    for i in range(len(datasets)):
        # przejcie po wszystkich punktach
        for j in range(datasets[i][1].shape[0]):
            if j==0:
                # zaznaczanie punktu dla pierwszego (SepalLengthCm) i drugiego (SepalWidthCm) parametru
                plt.scatter(datasets[i][1][j,0],datasets[i][1][j,1], color=colors[i], label=datasets[i][0]) 
            else:
                # zaznaczanie punktu dla pierwszego (SepalLengthCm) i drugiego (SepalWidthCm) parametru
                plt.scatter(datasets[i][1][j,0],datasets[i][1][j,1], color=colors[i])
    
    plt.grid()
    plt.legend()
    plt.show()
