def calculate_dot_product(v1,v2):
    dims = v1.shape[0]
    dot_product = 0
    
    for i in range(dims):
        dot_product = dot_product + v1[i]*v2[i]
    
    return dot_product

def vector_len(v1):
    dims = v1.shape[0]
    length = 0
    
    for i in range(dims):
        length = length + pow(v1[i],2)
    
    return length

def vector_normalization(v1):
    v1_len = vector_len(v1)
    arr = []
    
    for item in v1:
        arr.append(item/v1_len)
    
    return np.array(arr)

def shift(v1, theta):
    dims = v1.shape[0]
    v1_sum = 0
    
    for i in range(dims):
        v1_sum = v1_sum + pow(v1[i],2)
        
    return (theta/sqrt(v1_sum)) * (v1/(sqrt(v1_sum)))
