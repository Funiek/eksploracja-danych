def cov(x, y, x_avg, y_avg):
    elements_count = len(x) # iloÅ›Ä‡ elementÃ³w w tablicy
    elements_sum = 0
    
    for x,y in zip(x,y):
        elements_sum = elements_sum + ((x - x_avg)*(y - y_avg)) # sumowanie (x - Å›rednia_arytmetyczna_x) * (y - Å›rednia_arytmetyczna_y)
    
    return elements_sum / elements_count # suma dzielona przez liczbÄ™ elementÃ³w

def cov_for_dataset(cov_dataset):
    avgs = []
    matrix_shape = cov_dataset.shape[1] # wymiary_macierzy czyli liczba_kolumn_bioru_danych x liczba_kolumn_bioru_danych
    for i in range(matrix_shape):
        avgs.append(cov_dataset[:,i].mean()) # obliczanie Å›rednich dla kaÅ¼dego parametru
    
    matrix = []
    for i in range(matrix_shape):
        matrix_row = []
        for j in range(matrix_shape):
            #matrix_shape*i + j
            res = cov(cov_dataset[:,i],cov_dataset[:,j],avgs[i], avgs[j]) # wyznaczanie kowariancji dla kaÅ¼dej pary parametrÃ³w
            matrix_row.append(res) # budowanie wiersza macierzy
        matrix.append(matrix_row) # dodawanie wiersza macierzy do tablicy bazowej
    
    return np.matrix(matrix) # przeksztaÅ‚canie typu tablicy na macierz

def eigenvector_len(v1):
    dims = v1.shape[0] # iloÅ›Ä‡ elementÃ³w wektora
    length = 0
    
    for i in range(dims):
        val = v1[i].ravel().tolist()[0][0]
        length = length + pow(val,2) # sumowanie elementÃ³w podniesionych do kwadratu
        
    
    length = sqrt(length) # pierwiastkowanie wyniku sumy elementÃ³w wektora podniesionych do kwadratÃ³w
    
    return length # zwracanie dÅ‚ugoÅ›ci

def eigenvector_normalization(v1):
    eigenvalue = eigenvector_len(v1) # wylicznaie wartoÅ›ci wÅ‚asnej
    eigenvector = v1 / eigenvalue # wyliczanie wektora wÅ‚asnego
    
    return eigenvalue, eigenvector # zwrÃ³cenie wartoÅ›ci wÅ‚asnej i wektora wÅ‚asnego

def rearrange_matrix(eigenvectors):
    res = []

    for i in range(len(eigenvectors)):
        row = eigenvectors[i].ravel().tolist()[0]
        res.append(row)

    return np.matrix(res).T

def eigenvalues_eigenvectors(matrix, num_iterations):
    n = matrix.shape[0] # jeden z wymiarÃ³w macierzy, poniewaÅ¼ jest kwadratowa to wystarczy tylko jeden
    eigenvalues = [] # lista na wartoÅ›ci wÅ‚asne
    eigenvectors = [] # lista na wektory wÅ‚asne
    
    for i in range(n):
        # losowe dane wektora na poczÄ…tku
        eigenvector = np.random.rand(n,1)
    
        # algorytm iteracyjny wykonywany num_iterations razy
        for _ in range(num_iterations):
            eigenvector = np.dot(matrix, eigenvector) # mnoÅ¼enie macierzy kowariancji przez wektor 
            eigenvalue, eigenvector = eigenvector_normalization(eigenvector) # normalizacja wektora
            
        # usuwanie z macierzy znalezionej wartoÅ›ci wÅ‚asnej i wektora wÅ‚asnego
        # marix - wartoÅ›Ä‡ wÅ‚asna * iloczyn zewnÄ™trzny wektora wÅ‚asnego
        matrix = matrix - eigenvalue * np.multiply(eigenvector,eigenvector.T).T

        else:
            eigenvector = eigenvector * -1
            matrix = matrix - eigenvalue * np.multiply(eigenvector,eigenvector.T).T'''
        
        eigenvalues.append(eigenvalue) # dodanie znalezionej wartoÅ›ci wÅ‚asnej do listy
        eigenvectors.append(eigenvector) # dodanie znalezionego wektora wÅ‚asnego do listy
